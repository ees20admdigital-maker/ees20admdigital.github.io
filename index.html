// ========================================================== // APPS SCRIPT v4 – CON JSONP PARA GITHUB PAGES // ========================================================== var SHEET_NAME = "Asistencia EES20"; var STATS_SHEET = "Stats"; var ACTAS_SHEET = "Actas"; function doGet(e) { var action = e.parameter.action || "resumen"; var callback = e.parameter.callback || ""; // para JSONP var result; if (action === "resumen") result = resumen(); else if (action === "stats") result = getStats(); else if (action === "actas") result = getActas(); else result = {ok:false, error:"accion desconocida"}; var json = JSON.stringify(result); // Si viene con callback= lo devolvemos como JSONP if (callback) { return ContentService .createTextOutput(callback + "(" + json + ")") .setMimeType(ContentService.MimeType.JAVASCRIPT); } return ContentService .createTextOutput(json) .setMimeType(ContentService.MimeType.JSON); } function doPost(e) { var data = JSON.parse(e.postData.contents); var action = data.action || ""; if (action === "guardar") return guardar(data.rows); if (action === "guardar_stats") return guardarStats(data.stats); return resp({ok:false, error:"accion desconocida"}); } function resp(obj) { return ContentService.createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); } function guardar(rows) { try { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheetByName(SHEET_NAME); if (!sheet) sheet = ss.insertSheet(SHEET_NAME); var now = new Date().toISOString(); for (var i = 0; i < rows.length; i++) { var r = rows[i]; sheet.appendRow([r.curso, r.materia, r.fecha, r.nclase, r.nombre, r.estado, r.obs || "", now]); } return resp({ok:true}); } catch(ex) { return resp({ok:false, error: ex.toString()}); } } function resumen() { try { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheetByName(SHEET_NAME); if (!sheet) return {ok:true, rows:[]}; var data = sheet.getDataRange().getValues(); var rows = []; for (var i = 0; i < data.length; i++) { var r = data[i]; rows.push({ curso: r[0], materia: r[1], fecha: r[2], nclase: r[3], nombre: r[4], estado: r[5], acta: r[6], guardado:r[7] }); } return {ok:true, rows:rows}; } catch(ex) { return {ok:false, error: ex.toString()}; } } function getStats() { try { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheetByName(STATS_SHEET); if (!sheet) return {ok:true, rows:[]}; var data = sheet.getDataRange().getValues(); var rows = []; for (var i = 0; i < data.length; i++) { rows.push({curso:data[i][0], materia:data[i][1], fecha:data[i][2], stats:data[i][3]}); } return {ok:true, rows:rows}; } catch(ex) { return {ok:false, error: ex.toString()}; } } function getActas() { try { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheetByName(ACTAS_SHEET); if (!sheet) return {ok:true, rows:[]}; var data = sheet.getDataRange().getValues(); var rows = []; for (var i = 0; i < data.length; i++) { rows.push({curso:data[i][0], materia:data[i][1], nombre:data[i][2], fecha:data[i][3], acta:data[i][4]}); } return {ok:true, rows:rows}; } catch(ex) { return {ok:false, error: ex.toString()}; } } function guardarStats(stats) { try { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheetByName(STATS_SHEET); if (!sheet) sheet = ss.insertSheet(STATS_SHEET); sheet.appendRow([stats.curso, stats.materia, stats.fecha, JSON.stringify(stats)]); return resp({ok:true}); } catch(ex) { return resp({ok:false, error: ex.toString()}); } }
